SELECT DISTINCT
     TMP.IDPRODUTO,
     TMP.IDSUBPRODUTO,
     TMP.PERICMSAI,
     TMP.TIPOSITTRIBSAI,
     TMP.IDSITTRIBSAI,
     TMP.PERREDTRIBSAI,
     TMP.PERICMECF,
     CAST ('0' AS DECIMAL(5,2)) AS PERREDSUBTRIB,
     CAST ('0' AS DECIMAL(5,2)) AS PERICMSUBST,
     TMP.FLAGISS,
     TMP.UFORIGEM,
     TMP.VALPRECOPRODUTO,
     TMP.DTINIPROMOCAO,
     TMP.DTFIMPROMOCAO,
     TMP.VALPRECOPROMOCAO,
     TMP.IDEMPRESA,
     TMP.QTDMINPROMVAREJO,
     TMP.FLAGPROMOCAO,
     TMP.IDCODBARPROD,
     DBA.UF_UTF8_TO_WIN1252( LEFT(TMP.DESCRRESPRODUTO,40 )) AS DESCRRESPRODUTO,
     TMP.VALMULTIVENDAS,
     TMP.IDINFORMACAONUTRICIONAL,
     TMP.FLAGCESTABASICA,
     TMP.FLAGPRODUTOACABADO,
     TMP.NCM,
     TMP.IDSUBCATIPIRECEITA,
     TMP.IDLISTASERVICORECEITA,
     TMP.FLAGCOMPONENTE,
     TMP.IDPRODUTOVASILHAME,
     TMP.IDSUBPRODUTOVASILHAME,
     TMP.FLAGEXPBALANCA,
     TMP.TIPOBAIXAMESTRE,
     TMP.FLAGINATIVO,
     TMP.FLAGVASILHAME,
     TMP.EMBALAGEMSAIDA,
     TMP.VALGRAMASAIDA,
     TMP.IDSECAO,
     TMP.IDGRUPO,
     TMP.IDSUBGRUPO,
     TMP.IDDIVISAO,
     TMP.DIASVALIDADE,
     TMP.FLAGBLOQUEIAVENDA,
     TMP.FLAGEVIDENCIA,
     UF_BUSCATIPOPISCOFINS(TMP.IDCSTPISCOFINS) AS TIPOPISCOFINS,
     TMP.IDCSTPISCOFINS AS IDCSTPISCOFINS,
     COALESCE(CAST(UF_GETPERPIS(TMP.IDEMPRESA, 1300, TMP.IDSUBPRODUTO, 'S', TMP.IDCENARIOFISCAL) AS NUMERIC(7,4)), 0) AS PERPIS,
     COALESCE(CAST(UF_GETPERCOFINS(TMP.IDEMPRESA, 1300, TMP.IDSUBPRODUTO, 'S', TMP.IDCENARIOFISCAL) AS NUMERIC(7,4)), 0) AS PERCOFINS,
        CASE WHEN (( SELECT COALESCE(COI.IDNATUREZAPISCOFINS,0) AS IDNATUREZAPISCOFINS FROM DBA.OPERACAO_INTERNA COI WHERE COI.IDOPERACAO = 1300 ) > 0 ) THEN
                         ( SELECT COALESCE(COI.IDNATUREZAPISCOFINS,0) AS IDNATUREZAPISCOFINS FROM DBA.OPERACAO_INTERNA COI WHERE COI.IDOPERACAO = 1300 )
         ELSE
                        COALESCE(CAST(UF_GETCODNATUREZAPISCOFINS( TMP.IDSUBPRODUTO, TMP.IDCENARIOFISCAL) AS INTEGER), 1)
         END AS IDNATUREZAPISCOFINS,
     COALESCE(MAX_COLUMN_TIMESTAMP(MAX_COLUMN_TIMESTAMP(MAX_COLUMN_TIMESTAMP(TMP.CENFISDTALTERACAO, TMP.PROGRADTALTERACAO),PRODUTDTALTERACAO), TMP.PRECOPRODUTODTALTERACAO), '1900-01-01 00:00:00') AS DTALTERACAO,
     TMP.IDSUBPRODUTOEXTERNO,
     TMP.IDCENARIOFISCAL,
     TMP.PERCARGATRIBUTARIA,
     COALESCE(CAST(TMP.PERALIQFEDERAL AS NUMERIC(5,2)), 0) AS PERALIQFEDERAL,
     COALESCE(CAST(TMP.PERALIQESTADUAL AS NUMERIC(5,2)), 0) AS PERALIQESTADUAL,
     COALESCE(CAST(TMP.PERALIQMUNICIPAL AS NUMERIC(5,2)), 0) AS PERALIQMUNICIPAL,
     COALESCE((Select trim(dados) From config_entidade Where entidade = 'IBP' and chave1 = :RA_IDEMPRESA and chave2 = '2'), '') as CODIGOIBPT,
     CODCEST,
     TMP.FLAGSOLICITAPESAGEM,
     COALESCE(TMP.ICMS_FCEP, 0) as PERFCEP,
     COALESCE(TMP.PESOBRUTO, 0) AS PESOBRUTO,
     TMP.FLAGPRODUZIDOESCALANAORELEVANTE,
     MARCA.CNPJFABRICANTE,
        MARCA.FLAGNAOAPLICARACRESCIMOFORMA,
     TMP.IDCADEIAPRECO,
     TMP.IDCODBARPRODTRIB,
     TMP.FLAGLOTE,
     TMP.FLAGTINTA,
     (CASE WHEN TMP.TIPOSITTRIBSAI = 'T' AND COALESCE(TMP.PERREDTRIBSAI,0) > 0 THEN
        UF_GET_INFORMACAO_ADICIONAL( :RA_IDEMPRESA , 'R', TMP.NCM, TMP.CODCEST)
        WHEN TMP.IDSITTRIBSAI = 41 THEN
                UF_GET_INFORMACAO_ADICIONAL( :RA_IDEMPRESA , 'N', TMP.NCM, TMP.CODCEST)
        WHEN TMP.IDSITTRIBSAI = 61 THEN
                 (
                        SELECT
                                          CIAA.CODIGOINFOADICIONAL
                        FROM
                                          DBA.CONFIG_INFORMACAO_ADICIONAL_AUTOMATICA AS CIAA
                        LEFT JOIN
                                          DBA.CONFIG_INFORMACAO_ADICIONAL_AUTOMATICA_NCM AS CIAAN
                                                                 ON ( CIAA.IDINFADIAUTO = CIAAN.IDINFADIAUTO )
                        WHERE
                                          CIAA.IDEMPRESA = :RA_IDEMPRESA AND
                                          CIAA.IDCFOP = 5656 AND
                                          CIAA.TIPOSITTRIB = 'M' AND
                                          CIAA.FLAGENVIACODIGOBENEFICIO = 'T' AND
                                          ( CIAAN.NCM = TMP.NCM OR TRIM( CIAAN.NCM ) = TMP.NCM OR CIAAN.NCM IS NULL )
                        FETCH FIRST 1 ROWS ONLY
                 )
     ELSE
        UF_GET_INFORMACAO_ADICIONAL( :RA_IDEMPRESA , TMP.TIPOSITTRIBSAI, TMP.NCM, TMP.CODCEST)
     END) AS INFORMACAO_ADICIONAL,
     TMP.FLAGCONFERENCIAFRCX,
     DBA.UF_UTF8_TO_WIN1252( LEFT(TMP.DESCRICAOCOMPLETA,150)) AS DESCRICAOCOMPLETA,
     TMP.PERDESCONTOMAXIMO,
     TMP.FLAGCOMBUSTIVEL,
     TMP.GRAMATURA,
     TMP.IDCODIGOANP,
     DBA.UF_UTF8_TO_WIN1252( LEFT(TMP.DESCRICAOANP, 200)) AS DESCRICAOANP,
     TMP.PERGASNATURALGLP,
     TMP.PERGASNATURALGLPIMP,
         TMP.QTDMAXVENDAMES,
          (SELECT P.PATHTHUMBNAIL FROM PRODUTO_GRADE_IMAGENS P
         WHERE P.IDPRODUTO = TMP.IDPRODUTO AND P.IDSUBPRODUTO = TMP.IDSUBPRODUTO  AND
           COALESCE(P.PATHTHUMBNAIL, '') <> ''
                 ORDER BY P.FLAGIMAGEMPADRAO DESC, P.IDCODIMAGEM FETCH FIRST 1 ROWS ONLY) AS PATHTHUMBNAIL,
        TMP.REFERENCIA,
        TMP.PERIPISAIDA,
        TMP.FLAGBLOQUEIASEMESCANEAR,
           TMP.CLASSFISCAL,
                TMP.VALPRECOATACADO,
        TMP.GRAMATURAVENDA,
        TMP.QTDTOTALEMBALAGEM,
        TMP.PESOLIQUIDO,
        TMP.FLAGPERMITEDESCONTO,
        TMP.TIPOITEM,
        TMP.IDMARCAFABRICANTE,
        TMP.MODELO,
        TMP.FLAGRETENCAOPIS,
        TMP.FLAGRETENCAOCOFINS,
        TMP.FLAGRETENCAOIR,
        TMP.FLAGRETENCAOCONTRSOCIAL,
        CFMSG.IDMSGSAI AS IDMSG
FROM
     (
     SELECT
              CAST ('1' AS INTEGER) AS VALIDACAO,
              PRODUTOS.IDCENARIOFISCAL,
              PRODUTOS.IDPRODUTO,
              PRODUTOS.IDSUBPRODUTO,
              PRODUTOS.PERICMSAI,
              PRODUTOS.TIPO_ANTECIPACAO,
              UF_GETTIPOSITTRIB(PRODUTOS.IDSITTRIBSAI, PRODUTOS.TIPO_ANTECIPACAO) AS TIPOSITTRIBSAI,
              PRODUTOS.IDSITTRIBSAI,
              PRODUTOS.PERREDTRIBSAI,
              PRODUTOS.PERICMECF,
              PRODUTOS.FLAGISS,
              PRODUTOS.UFORIGEM,
              PRODUTOS.CENFISDTALTERACAO,
              PRODUTOS.PROGRADTALTERACAO,
              PRODUTOS.PRODUTDTALTERACAO,
              CASE WHEN (PRODUTOS.FLAGEVIDENCIAP = 'T' OR PRODUTOS.FLAGEVIDENCIAPG = 'T' OR PRODUTOS.FLAGEVIDENCIAC = 'T') OR PPP.FLAGEVIDENCIA = 'T' THEN
                            'T'
              ELSE
                            'F'
              END AS FLAGEVIDENCIA,
              COALESCE(CAST(UF_GETCSTPISCOFINS(PPP.IDEMPRESA, 1300, PRODUTOS.IDSUBPRODUTO,'S', PRODUTOS.IDCENARIOFISCAL) AS INTEGER), 1) AS IDCSTPISCOFINS,
              COALESCE(PPP.VALPRECOVAREJO,0) AS VALPRECOPRODUTO,
              CAST( CAST(COALESCE(PPP.DTINIPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 00:00:00' AS TIMESTAMP) AS DTINIPROMOCAO,
              CAST( CAST(COALESCE(PPP.DTFIMPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 23:59:59' AS TIMESTAMP) AS DTFIMPROMOCAO,
              PPP.DTINIPROMOCAOVAR,
              PPP.DTFIMPROMOCAOVAR,
              PPP.VALPROMVAREJO AS VALPRECOPROMOCAO,
              PPP.IDEMPRESA,
              PPP.DTALTERACAO AS PRECOPRODUTODTALTERACAO,
              PPP.QTDMINPROMVAREJO,
              PPP.PERCARGATRIBUTARIA,
              PPP.PERALIQNACFEDERAL AS PERALIQFEDERAL,
              PPP.PERALIQESTADUAL,
              PPP.PERALIQMUNICIPAL,
              PRODUTOS.FLAGPROMOCAO,
              PRODUTOS.IDCODBARPROD,
              PRODUTOS.DESCRRESPRODUTO,
              PRODUTOS.VALMULTIVENDAS,
              PRODUTOS.IDINFORMACAONUTRICIONAL,
              PRODUTOS.FLAGCESTABASICA,
              PRODUTOS.FLAGPRODUTOACABADO,
              PRODUTOS.NCM,
              PRODUTOS.IDSUBCATIPIRECEITA,
              PRODUTOS.IDLISTASERVICORECEITA,
              PRODUTOS.FLAGCOMPONENTE,
              PRODUTOS.IDPRODUTOVASILHAME,
              PRODUTOS.IDSUBPRODUTOVASILHAME,
              PRODUTOS.FLAGEXPBALANCA,
              PRODUTOS.TIPOBAIXAMESTRE,
              PRODUTOS.FLAGINATIVO,
              PRODUTOS.FLAGVASILHAME,
              PRODUTOS.EMBALAGEMSAIDA,
              PRODUTOS.VALGRAMASAIDA,
              PRODUTOS.IDSECAO,
              PRODUTOS.IDGRUPO,
              PRODUTOS.IDSUBGRUPO,
              PRODUTOS.IDDIVISAO,
              PRODUTOS.DTALTERACAO,
              PRODUTOS.DIASVALIDADE,
              PRODUTOS.FLAGBLOQUEIAVENDA,
              PRODUTOS.IDSUBPRODUTOEXTERNO,
              PRODUTOS.CODCEST,
              PRODUTOS.FLAGSOLICITAPESAGEM,
              PRODUTOS.ICMS_FCEP,
              PRODUTOS.PESOBRUTO,
              PRODUTOS.FLAGPRODUZIDOESCALANAORELEVANTE,
              PRODUTOS.IDMARCAFABRICANTE,
              PRODUTOS.IDCADEIAPRECO,
              PRODUTOS.IDCODBARPRODTRIB,
              PRODUTOS.FLAGLOTE,
              PRODUTOS.FLAGTINTA,
              PRODUTOS.FLAGCONFERENCIAFRCX,
              PRODUTOS.DESCRICAOCOMPLETA,
              DESCONTO_MAXIMO_PRODUTO_EMP.PERDESCONTOMAXIMO,
              COALESCE( PRODUTOS.FLAGCOMBUSTIVEL, 'F') AS FLAGCOMBUSTIVEL,
              COALESCE((SELECT GRAMATURA
                        FROM PRODUTO_GRADE_UNIDADE_TRIBUTARIA PGUNT
                        WHERE PGUNT.IDPRODUTO = PRODUTOS.IDPRODUTO AND
                             PGUNT.IDSUBPRODUTO = PRODUTOS.IDSUBPRODUTO
                             FETCH FIRST 1 ROWS ONLY), 0) AS GRAMATURA,
              COALESCE(PRODUTOS.IDCODIGOANP, '') AS IDCODIGOANP,
              CASE WHEN COALESCE(IDCODIGOANP, '') <> '' THEN (SELECT COALESCE(DESCRICAO, 'CODANP SEM DESCRICAO')
                                                              FROM CODIGO_PRODUTO_ANP
                                                              WHERE IDCODIGOANP = PRODUTOS.IDCODIGOANP
                                                              FETCH FIRST 1 ROWS ONLY)  ELSE  '' END AS DESCRICAOANP,
              COALESCE(PRODUTOS.PERGASNATURALGLP, 0) as PERGASNATURALGLP,
              COALESCE(PRODUTOS.PERGASNATURALGLPIMP, 0) as PERGASNATURALGLPIMP,
              PRODUTOS.QTDMAXVENDAMES,
              PRODUTOS.REFERENCIA,
              PRODUTOS.PERIPISAIDA,
              PRODUTOS.FLAGBLOQUEIASEMESCANEAR,
              PRODUTOS.CLASSFISCAL,
              COALESCE(PPP.VALPRECOATACADO,0) AS VALPRECOATACADO,
              PRODUTOS.GRAMATURAVENDA,
              PRODUTOS.QTDTOTALEMBALAGEM,
              PRODUTOS.PESOLIQUIDO,
              PRODUTOS.FLAGPERMITEDESCONTO,
              PRODUTOS.TIPOITEM,
              PRODUTOS.MODELO,
              PRODUTOS.FLAGRETENCAOPIS,
              PRODUTOS.FLAGRETENCAOCOFINS,
              PRODUTOS.FLAGRETENCAOIR,
              PRODUTOS.FLAGRETENCAOCONTRSOCIAL
     FROM
            (
              SELECT
                            EMPRESA.IDEMPRESA,
                            PTCV.IDCENARIOFISCAL,
                            PG.IDPRODUTO,
                            PG.IDSUBPRODUTO,
                            PTCV.PERICMSAI,
                            PTCV.TIPO_ANTECIPACAO,
                            PTCV.IDSITTRIBSAI,
                            PTCV.PERREDTRIBSAI,
                            PTCV.PERICMECF,
                            PTCV.UFORIGEM,
                            PTCV.CENFISDTALTERACAO,
                            'F' AS FLAGPROMOCAO,
                            PTCV.FLAGEVIDENCIA AS FLAGEVIDENCIAC,
                            PTCV.PERFCEP as ICMS_FCEP,
                            P.FLAGEVIDENCIA AS FLAGEVIDENCIAP,
                            PG.FLAGEVIDENCIA AS FLAGEVIDENCIAPG,
                            P.FLAGISS,
                            COALESCE(PG.DTALTERACAO, '1900-01-01 00:00:00') AS PROGRADTALTERACAO,
                            COALESCE(P.DTALTERACAO, '1900-01-01 00:00:00') AS PRODUTDTALTERACAO,
                            PG.IDCODBARPROD,
                            PG.DESCRRESPRODUTO,
                            COALESCE(PG.VALMULTIVENDAS, 1) AS VALMULTIVENDAS,
                            PG.IDINFORMACAONUTRICIONAL,
                            PG.FLAGCESTABASICA,
                            PG.FLAGPRODUTOACABADO,
                            PG.NCM,
                            PG.IDSUBCATIPIRECEITA,
                            PG.IDLISTASERVICORECEITA,
                            PG.FLAGCOMPONENTE,
                            PG.IDPRODUTOVASILHAME,
                            PG.IDSUBPRODUTOVASILHAME,
                            PG.FLAGEXPBALANCAGRADE AS FLAGEXPBALANCA,
                            P.TIPOBAIXAMESTRE,
                            COALESCE(PG.FLAGINATIVO, 'F') AS FLAGINATIVO,
                            COALESCE(PG.FLAGVASILHAME, 'F') AS FLAGVASILHAME,
                            P.EMBALAGEMSAIDA,
                            P.VALGRAMASAIDA,
                            P.IDSECAO,
                            P.IDGRUPO,
                            P.IDSUBGRUPO,
                            P. IDDIVISAO,
                            CASE WHEN P.DTALTERACAO > PG.DTALTERACAO THEN
                                     PG.DTALTERACAO
                            ELSE
                                     P.DTALTERACAO
                            END AS DTALTERACAO,
                            CASE WHEN PG.DIASVALIDADE > 0 THEN
                                     PG.DIASVALIDADE
                            ELSE
                                     P.DIASVALIDADE
                            END AS DIASVALIDADE,
                            COALESCE(PG.FLAGBLOQUEIAVENDA, 'F') AS FLAGBLOQUEIAVENDA,
                            PG.IDSUBPRODUTOEXTERNO,
                            PG.CODCEST,
                            P.FLAGSOLICITAPESAGEM,
                            PG.PESOBRUTO,
                            PG.FLAGPRODUZIDOESCALANAORELEVANTE,
                            P.IDMARCAFABRICANTE,
                            PG.IDCADEIAPRECO,
                            PG.IDCODBARPRODTRIB,
                            PG.FLAGLOTE,
                            PG.FLAGTINTA,
                            PG.FLAGCONFERENCIAFRCX,
                            (P.DESCRCOMPRODUTO || ' ' || COALESCE(PG.SUBDESCRICAO,'')) AS DESCRICAOCOMPLETA,
                            COALESCE( PG.FLAGCOMBUSTIVEL, 'F') AS FLAGCOMBUSTIVEL,
                            PG.IDCODIGOANP,
                            PG.PERGASNATURALGLP,
                            PG.PERGASNATURALGLPIMP,
                                                 PG.QTDMAXVENDAMES,
                                                 COALESCE(PG.REFERENCIA, '') AS REFERENCIA,
                            PG.PERIPIEQINDUSTRIAL AS PERIPISAIDA,
                            PG.FLAGBLOQUEIASEMESCANEAR AS FLAGBLOQUEIASEMESCANEAR,
                                           P.CLASSFISCAL AS CLASSFISCAL,
                            PG.GRAMATURAVENDA,
                            PG.QTDTOTALEMBALAGEM,
                            PG.PESOLIQUIDO,
                            PG.FLAGPERMITEDESCONTO ,
                            PG.TIPOITEM,
                            P.MODELO,
                            P.FLAGRETENCAOPIS,
                            P.FLAGRETENCAOCOFINS,
                            P.FLAGRETENCAOIR,
                            P.FLAGRETENCAOCONTRSOCIAL
              FROM
                            PRODUTO_GRADE PG
                            JOIN PRODUTO P ON (
                                     PG.IDPRODUTO = P.IDPRODUTO
                            ),
                             PRODUTO_TRIBUTACAO_CENARIO_VIEW PTCV,
                                                            EMPRESA
                                                            JOIN ATIVIDADE_TIPO_ATIVIDADE ON (
                                                                     ATIVIDADE_TIPO_ATIVIDADE.IDATIVIDADE = EMPRESA.IDATIVIDADE
                                                            )
                                                            JOIN CENARIO_FISCAL_PADRAO_PRODUTO_VW ON (
                                                                     CENARIO_FISCAL_PADRAO_PRODUTO_VW.IDEMPRESA = EMPRESA.IDEMPRESA AND
                                                                     CENARIO_FISCAL_PADRAO_PRODUTO_VW.OPERACAO = 'S' AND
                                                                     CENARIO_FISCAL_PADRAO_PRODUTO_VW.IDTIPOOPERACAO = 96 and
                                                                                                                             CENARIO_FISCAL_PADRAO_PRODUTO_VW.UF = EMPRESA.UF
                                                            )
              WHERE
                            P.TIPOBAIXAMESTRE <> 'C' AND
                            ('F' = :RA_COMPLETA OR PG.FLAGINATIVO <> :RA_COMPLETA OR
                        (SELECT COALESCE(V_VARCHAR, 'F') FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTOVALOR = (
                SELECT MAX(IDATRIBUTOVALOR) FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTO = 273 AND IDEMPRESA = :RA_IDEMPRESA)) = 'T') AND
                            (PTCV.UFORIGEM = :RA_UF OR PTCV.UFORIGEM = '')                      AND
                             PTCV.IDPRODUTO = PG.IDPRODUTO AND
                             PTCV.IDSUBPRODUTO = PG.IDSUBPRODUTO and
                            EMPRESA.IDEMPRESA                   = :RA_IDEMPRESA                 AND
                            ATIVIDADE_TIPO_ATIVIDADE.FLAGPADRAO = 'T'                           AND
                            PTCV.IDTIPOATIVIDADEORIGEM          = ATIVIDADE_TIPO_ATIVIDADE.IDTIPOATIVIDADE AND
                            PTCV.IDREGIMEESPECIALORIGEM         = EMPRESA.IDREGIMEESPECIAL      AND
                            PTCV.UFDESTINO                      = EMPRESA.UF                    AND
                            PTCV.TIPOREGIMETRIBFEDERALORIGEM    = EMPRESA.TIPOREGIMETRIBFEDERAL AND
                            PTCV.IDORIGEMPRODUTO                = CENARIO_FISCAL_PADRAO_PRODUTO_VW.IDORIGEMPRODUTO AND
                            CENARIO_FISCAL_PADRAO_PRODUTO_VW.IDPRODUTO      = PTCV.IDPRODUTO    AND
                            CENARIO_FISCAL_PADRAO_PRODUTO_VW.IDSUBPRODUTO   = PTCV.IDSUBPRODUTO
              ) AS PRODUTOS
              JOIN POLITICA_PRECO_PRODUTO PPP ON (
                            PRODUTOS.IDPRODUTO = PPP.IDPRODUTO AND
                            PRODUTOS.IDSUBPRODUTO = PPP.IDSUBPRODUTO AND
                            PRODUTOS.IDEMPRESA = PPP.IDEMPRESA
              )
              LEFT OUTER JOIN DBA.DESCONTO_MAXIMO_PRODUTO_EMP DESCONTO_MAXIMO_PRODUTO_EMP ON (
                            DESCONTO_MAXIMO_PRODUTO_EMP.IDEMPRESA = PPP.IDEMPRESA AND
                            DESCONTO_MAXIMO_PRODUTO_EMP.IDPRODUTO = PPP.IDPRODUTO AND
                            DESCONTO_MAXIMO_PRODUTO_EMP.IDSUBPRODUTO = PPP.IDSUBPRODUTO
              )
              LEFT JOIN LATERAL (
                   SELECT DISTINCT
                       'T' AS FLAGMIX, IDPRODUTO, IDSUBPRODUTO
                   FROM
                       DBA.PRODUTO_EMPRESA AS PRODUTO_EMPRESA
                   WHERE
                       PPP.IDPRODUTO = PRODUTO_EMPRESA.IDPRODUTO AND
                       PPP.IDSUBPRODUTO = PRODUTO_EMPRESA.IDSUBPRODUTO
               ) AS MIX ON (
                   MIX.IDPRODUTO    = PPP.IDPRODUTO AND
                   MIX.IDSUBPRODUTO = PPP.IDSUBPRODUTO
               )
               LEFT JOIN DBA.PRODUTO_EMPRESA PE ON (
                   PE.IDEMPRESA    = PPP.IDEMPRESA    AND
                   PE.IDPRODUTO    = PPP.IDPRODUTO    AND
                   PE.IDSUBPRODUTO = PPP.IDSUBPRODUTO
               )
               LEFT JOIN
               ( SELECT IDEMPRESA, COALESCE(V_VARCHAR, 'F') AS FLAGPRODUTOEMPRESA
                 FROM CONFIG_ATRIBUTO_VALORES
                 WHERE IDATRIBUTO = 159 AND IDEMPRESA = :RA_IDEMPRESA
               ) AS CAV ON (CAV.IDEMPRESA = PPP.IDEMPRESA)
     WHERE
              ( PRODUTOS.PROGRADTALTERACAO >= :RA_DTMONITOR OR
                PRODUTOS.PRODUTDTALTERACAO >= :RA_DTMONITOR OR
                PPP.DTALTERACAO IS NULL OR
                PPP.DTALTERACAO >= :RA_DTMONITOR OR
                PRODUTOS.CENFISDTALTERACAO >= :RA_DTMONITOR) AND
               ( (CAV.FLAGPRODUTOEMPRESA = 'F' OR CAV.FLAGPRODUTOEMPRESA IS NULL) OR
                 (CAV.FLAGPRODUTOEMPRESA = 'T' AND (MIX.FLAGMIX IS NULL OR PE.FLAGPERMITEVENDA = 'T' OR PE.FLAGPERMITEVENDA IS NULL))) AND
                 PRODUTOS.TIPOITEM NOT IN('07', '08')
     UNION ALL

     SELECT
              CAST ('0' AS INTEGER) AS VALIDACAO,
              CAST ('0' AS INTEGER) AS IDCENARIOFISCAL,
              PG.IDPRODUTO,
              PG.IDSUBPRODUTO,
              CAST ('0' AS DECIMAL(16,2)) AS PERICMSAI,
              CAST ('0' AS INTEGER) AS TIPO_ANTECIPACAO,
              CAST ('I' AS CHAR) AS TIPOSITTRIBSAI,
              CAST ('40' AS INTEGER) AS IDSITTRIBSAI,
              CAST ('0' AS DECIMAL(16,2)) AS PERREDTRIBSAI,
              CAST ('0' AS DECIMAL(16,2)) AS PERICMECF,
              CAST ('F' AS CHAR) AS FLAGISS,
              CAST (:RA_UF AS CHAR(2)) AS UFORIGEM,
              CAST (:RA_DTMONITOR AS TIMESTAMP) AS CENFISDTALTERACAO,
              PG.DTALTERACAO AS PROGRADTALTERACAO,
              P.DTALTERACAO AS PRODUTDTALTERACAO,
              CASE WHEN P.FLAGEVIDENCIA = 'T' OR PG.FLAGEVIDENCIA= 'T' OR PPP.FLAGEVIDENCIA = 'T' THEN
                            'T'
              ELSE
                            'F'
              END AS FLAGEVIDENCIA,
              COALESCE(CAST(UF_GETCSTPISCOFINS(PPP.IDEMPRESA, 1300, PG.IDSUBPRODUTO,'S', NULL) AS INTEGER), 1) AS IDCSTPISCOFINS,
              COALESCE(PPP.VALPRECOVAREJO,0)VALPRECOPRODUTO,
              CAST( CAST(COALESCE(PPP.DTINIPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 00:00:00' AS TIMESTAMP) AS DTINIPROMOCAO,
              CAST( CAST(COALESCE(PPP.DTFIMPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 23:59:59' AS TIMESTAMP) AS DTFIMPROMOCAO,
              PPP.DTINIPROMOCAOVAR,
              PPP.DTFIMPROMOCAOVAR,
              PPP.VALPROMVAREJO AS VALPRECOPROMOCAO,
              PPP.IDEMPRESA,
              PPP.DTALTERACAO AS PRECOPRODUTODTALTERACAO,
              PPP.QTDMINPROMVAREJO,
              PPP.PERCARGATRIBUTARIA,
              PPP.PERALIQNACFEDERAL AS PERALIQFEDERAL,
              PPP.PERALIQESTADUAL,
              PPP.PERALIQMUNICIPAL,
              'F'AS FLAGPROMOCAO,
              PG.IDCODBARPROD,
              PG.DESCRRESPRODUTO,
              COALESCE(PG.VALMULTIVENDAS, 1) AS VALMULTIVENDAS,
              PG.IDINFORMACAONUTRICIONAL,
              PG.FLAGCESTABASICA,
              PG.FLAGPRODUTOACABADO,
              PG.NCM,
              PG.IDSUBCATIPIRECEITA,
              PG.IDLISTASERVICORECEITA,
              PG.FLAGCOMPONENTE,
              PG.IDPRODUTOVASILHAME,
              PG.IDSUBPRODUTOVASILHAME,
              PG.FLAGEXPBALANCAGRADE AS FLAGEXPBALANCA,
              P.TIPOBAIXAMESTRE,
              COALESCE(PG.FLAGINATIVO, 'F') AS FLAGINATIVO,
              COALESCE(PG.FLAGVASILHAME, 'F') AS FLAGVASILHAME,
              P.EMBALAGEMSAIDA,
              P.VALGRAMASAIDA,
              P.IDSECAO,
              P.IDGRUPO,
              P.IDSUBGRUPO,
              P. IDDIVISAO,
              CASE WHEN P.DTALTERACAO > PG.DTALTERACAO THEN
                            P.DTALTERACAO
              ELSE
                            PG.DTALTERACAO
              END AS DTALTERACAO,
              CASE WHEN PG.DIASVALIDADE > 0 THEN
                            PG.DIASVALIDADE
              ELSE
                            P.DIASVALIDADE
              END AS DIASVALIDADE,
              COALESCE(PG.FLAGBLOQUEIAVENDA, 'F') AS FLAGBLOQUEIAVENDA,
              PG.IDSUBPRODUTOEXTERNO,
              PG.CODCEST,
              P.FLAGSOLICITAPESAGEM,
              CAST ('0' AS DECIMAL(16,2)) AS ICMS_FCEP,
              PG.PESOBRUTO,
              PG.FLAGPRODUZIDOESCALANAORELEVANTE,
              P.IDMARCAFABRICANTE,
              PG.IDCADEIAPRECO,
              PG.IDCODBARPRODTRIB,
              PG.FLAGLOTE,
              PG.FLAGTINTA,
              PG.FLAGCONFERENCIAFRCX,
              (P.DESCRCOMPRODUTO || ' ' || COALESCE(PG.SUBDESCRICAO,'')) AS DESCRICAOCOMPLETA,
              DESCONTO_MAXIMO_PRODUTO_EMP.PERDESCONTOMAXIMO,
              COALESCE( PG.FLAGCOMBUSTIVEL, 'F') AS FLAGCOMBUSTIVEL,
              COALESCE((SELECT GRAMATURA
                        FROM PRODUTO_GRADE_UNIDADE_TRIBUTARIA PGUNT
                        WHERE PGUNT.IDPRODUTO = PG.IDPRODUTO AND
                              PGUNT.IDSUBPRODUTO = PG.IDSUBPRODUTO
                              FETCH FIRST 1 ROWS ONLY), 0) AS GRAMATURA,
              COALESCE(IDCODIGOANP, '') AS IDCODIGOANP,
              CASE WHEN COALESCE(PG.IDCODIGOANP, '') <> '' THEN (SELECT COALESCE(DESCRICAO, 'CODANP SEM DESCRICAO')
                                                                 FROM CODIGO_PRODUTO_ANP
                                                                 WHERE IDCODIGOANP = PG.IDCODIGOANP
                                                                 FETCH FIRST 1 ROWS ONLY)
              ELSE  '' END AS DESCRICAOANP,
              COALESCE(PG.PERGASNATURALGLP, 0) as PERGASNATURALGLP,
              COALESCE(PG.PERGASNATURALGLPIMP, 0) as PERGASNATURALGLPIMP,
                        PG.QTDMAXVENDAMES,
                        COALESCE(PG.REFERENCIA, '') AS REFERENCIA,
                        PG.PERIPIEQINDUSTRIAL AS PERIPISAIDA,
                        PG.FLAGBLOQUEIASEMESCANEAR AS FLAGBLOQUEIASEMESCANEAR,
                                    P.CLASSFISCAL AS CLASSFISCAL,
                                        COALESCE(PPP.VALPRECOATACADO,0) AS VALPRECOATACADO,
                        PG.GRAMATURAVENDA,
                        PG.QTDTOTALEMBALAGEM,
                        PG.PESOLIQUIDO,
                        PG.FLAGPERMITEDESCONTO ,
                        PG.TIPOITEM,
                                P.MODELO,
                                P.FLAGRETENCAOPIS,
                                P.FLAGRETENCAOCOFINS,
                                P.FLAGRETENCAOIR,
                                P.FLAGRETENCAOCONTRSOCIAL
             FROM
              POLITICA_PRECO_PRODUTO PPP
              JOIN PRODUTO_GRADE PG ON (
                            PG.IDPRODUTO = PPP.IDPRODUTO AND
                            PG.IDSUBPRODUTO = PPP.IDSUBPRODUTO
              )
              JOIN PRODUTO P ON (
                            P.IDPRODUTO = PG.IDPRODUTO
              )
              LEFT OUTER JOIN DBA.DESCONTO_MAXIMO_PRODUTO_EMP DESCONTO_MAXIMO_PRODUTO_EMP ON (
                            DESCONTO_MAXIMO_PRODUTO_EMP.IDEMPRESA = PPP.IDEMPRESA AND
                            DESCONTO_MAXIMO_PRODUTO_EMP.IDPRODUTO = PPP.IDPRODUTO AND
                            DESCONTO_MAXIMO_PRODUTO_EMP.IDSUBPRODUTO = PPP.IDSUBPRODUTO
              )
              LEFT JOIN LATERAL (
                   SELECT DISTINCT
                       'T' AS FLAGMIX, IDPRODUTO, IDSUBPRODUTO
                   FROM
                       DBA.PRODUTO_EMPRESA AS PRODUTO_EMPRESA
                   WHERE
                       PPP.IDPRODUTO = PRODUTO_EMPRESA.IDPRODUTO AND
                       PPP.IDSUBPRODUTO = PRODUTO_EMPRESA.IDSUBPRODUTO
               ) AS MIX ON (
                   MIX.IDPRODUTO    = PPP.IDPRODUTO AND
                   MIX.IDSUBPRODUTO = PPP.IDSUBPRODUTO
               )
               LEFT JOIN DBA.PRODUTO_EMPRESA PE ON (
                   PE.IDEMPRESA    = PPP.IDEMPRESA       AND
                   PE.IDPRODUTO    = PPP.IDPRODUTO    AND
                   PE.IDSUBPRODUTO = PPP.IDSUBPRODUTO
               )
               LEFT JOIN
               ( SELECT IDEMPRESA, COALESCE(V_VARCHAR, 'F') AS FLAGPRODUTOEMPRESA
                 FROM CONFIG_ATRIBUTO_VALORES
                 WHERE IDATRIBUTO = 159 AND IDEMPRESA = :RA_IDEMPRESA
               ) AS CAV ON (CAV.IDEMPRESA = PPP.IDEMPRESA)

     WHERE
              P.TIPOBAIXAMESTRE = 'C' AND
              PPP.IDEMPRESA = :RA_IDEMPRESA AND
              ('F' = :RA_COMPLETA OR PG.FLAGINATIVO <> :RA_COMPLETA OR
                        (SELECT COALESCE(V_VARCHAR, 'F') FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTOVALOR = (
                SELECT MAX(IDATRIBUTOVALOR) FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTO = 273 AND IDEMPRESA = :RA_IDEMPRESA)) = 'T') AND
              ( COALESCE(PG.DTALTERACAO, '1900-01-01 00:00:00') >= :RA_DTMONITOR OR
                COALESCE(P.DTALTERACAO, '1900-01-01 00:00:00') >= :RA_DTMONITOR OR
                COALESCE(PPP.DTALTERACAO, '1900-01-01 00:00:00') >= :RA_DTMONITOR) AND
                ( (CAV.FLAGPRODUTOEMPRESA = 'F' OR CAV.FLAGPRODUTOEMPRESA IS NULL) OR
                  (CAV.FLAGPRODUTOEMPRESA = 'T' AND (MIX.FLAGMIX IS NULL OR PE.FLAGPERMITEVENDA = 'T' OR PE.FLAGPERMITEVENDA IS NULL))) AND
                  PG.TIPOITEM NOT IN('07', '08')
      ) AS TMP
     LEFT OUTER JOIN DBA.MARCA MARCA ON (MARCA.IDMARCAFABRICANTE = TMP.IDMARCAFABRICANTE)
     LEFT OUTER JOIN DBA.CENARIO_FISCAL_MENSAGEM CFMSG ON (CFMSG.IDCENARIOFISCAL = TMP.IDCENARIOFISCAL)

ORDER BY 1, 2
