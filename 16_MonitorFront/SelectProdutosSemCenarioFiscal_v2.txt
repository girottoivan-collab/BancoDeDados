WITH EMPRESAS_BASE AS (
    SELECT
        EMPRESA.IDEMPRESA,
        EMPRESA.UF
    FROM
        DBA.EMPRESA EMPRESA
),
PRODUTOS AS (
    SELECT
        EMPRESAS_BASE.IDEMPRESA,
        EMPRESAS_BASE.UF,
        PRODUTO_TRIBUTACAO_ESTADO.IDPRODUTO,
        PRODUTO_TRIBUTACAO_ESTADO.IDSUBPRODUTO
    FROM
        DBA.PRODUTO_TRIBUTACAO_ESTADO PRODUTO_TRIBUTACAO_ESTADO
        JOIN EMPRESAS_BASE ON (
            EMPRESAS_BASE.UF = PRODUTO_TRIBUTACAO_ESTADO.UF
        )
    WHERE
        PRODUTO_TRIBUTACAO_ESTADO.UFORIGEM = EMPRESAS_BASE.UF AND
        COALESCE(PRODUTO_TRIBUTACAO_ESTADO.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR

    UNION

    SELECT
        EMPRESAS_BASE.IDEMPRESA,
        EMPRESAS_BASE.UF,
        PRODUTOS_VIEW.IDPRODUTO,
        PRODUTOS_VIEW.IDSUBPRODUTO
    FROM
        DBA.GRUPO_TRIBUTACAO_ESTADO GRUPO_TRIBUTACAO_ESTADO
        JOIN DBA.PRODUTOS_VIEW PRODUTOS_VIEW ON (
            PRODUTOS_VIEW.IDGRUPO = GRUPO_TRIBUTACAO_ESTADO.IDGRUPO
        )
        JOIN EMPRESAS_BASE ON (
            EMPRESAS_BASE.UF = GRUPO_TRIBUTACAO_ESTADO.UF
        )
    WHERE
        GRUPO_TRIBUTACAO_ESTADO.UFORIGEM = EMPRESAS_BASE.UF AND
        COALESCE(GRUPO_TRIBUTACAO_ESTADO.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR

    UNION

    SELECT
    DISTINCT
        EMPRESAS_BASE.IDEMPRESA,
        EMPRESAS_BASE.UF,
        PRODUTO_TRIBUTACAO.IDPRODUTO,
        PRODUTO_TRIBUTACAO.IDSUBPRODUTO
    FROM
        DBA.PRODUTO_TRIBUTACAO PRODUTO_TRIBUTACAO
        JOIN EMPRESAS_BASE ON (
            EMPRESAS_BASE.UF = PRODUTO_TRIBUTACAO.UF
        )
    WHERE
        COALESCE(PRODUTO_TRIBUTACAO.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR

    UNION

    SELECT
        EMPRESAS_BASE.IDEMPRESA,
        EMPRESAS_BASE.UF,
        PRODUTOS_VIEW.IDPRODUTO,
        PRODUTOS_VIEW.IDSUBPRODUTO
    FROM
        DBA.GRUPO_TRIBUTACAO GRUPO_TRIBUTACAO
        JOIN DBA.PRODUTOS_VIEW PRODUTOS_VIEW ON (
            PRODUTOS_VIEW.IDGRUPO = GRUPO_TRIBUTACAO.IDGRUPO
        )
        JOIN EMPRESAS_BASE ON (
            EMPRESAS_BASE.UF = GRUPO_TRIBUTACAO.UF
        )
    WHERE
        COALESCE(GRUPO_TRIBUTACAO.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR

    UNION

    SELECT
        EMPRESAS_BASE.IDEMPRESA,
        EMPRESAS_BASE.UF,
        PRODUTO_GRADE.IDPRODUTO,
        PRODUTO_GRADE.IDSUBPRODUTO
    FROM
        EMPRESAS_BASE
        CROSS JOIN DBA.PRODUTO PRODUTO
        JOIN DBA.PRODUTO_GRADE PRODUTO_GRADE ON (
            PRODUTO_GRADE.IDPRODUTO = PRODUTO.IDPRODUTO
        )
        JOIN DBA.GRUPO GRUPO ON (
            PRODUTO.IDGRUPO = GRUPO.IDGRUPO
        )
    WHERE
        COALESCE(PRODUTO_GRADE.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR OR
        COALESCE(PRODUTO.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR OR
        COALESCE(GRUPO.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR

    UNION

    SELECT
        EMPRESAS_BASE.IDEMPRESA,
        EMPRESAS_BASE.UF,
        PPP.IDPRODUTO,
        PPP.IDSUBPRODUTO
    FROM
        DBA.POLITICA_PRECO_PRODUTO PPP
        JOIN EMPRESAS_BASE ON (
            EMPRESAS_BASE.IDEMPRESA = PPP.IDEMPRESA
        )
    WHERE
        COALESCE(PPP.DTALTERACAO, CAST('1900-01-01' AS TIMESTAMP)) >= :RA_DTMONITOR
)

        CASE WHEN (( SELECT COALESCE(COI.IDNATUREZAPISCOFINS,0) AS IDNATUREZAPISCOFINS FROM DBA.OPERACAO_INTERNA COI WHERE COI.IDOPERACAO = 1300 ) > 0 ) THEN
                         ( SELECT COALESCE(COI.IDNATUREZAPISCOFINS,0) AS IDNATUREZAPISCOFINS FROM DBA.OPERACAO_INTERNA COI WHERE COI.IDOPERACAO = 1300 )
         ELSE
                        COALESCE(CAST(UF_GETCODNATUREZAPISCOFINS( TMP.IDSUBPRODUTO, TMP.IDCENARIOFISCAL) AS INTEGER), 1)
         END AS IDNATUREZAPISCOFINS,

    TMP.FLAGSOLICITAPESAGEM,
    TMP.PERFCEP,

        MARCA.FLAGNAOAPLICARACRESCIMOFORMA,
    TMP.IDCADEIAPRECO,
    TMP.IDCODBARPRODTRIB,
    TMP.FLAGLOTE,
    TMP.FLAGTINTA,
   (CASE WHEN TMP.TIPOSITTRIBSAI = 'T' AND COALESCE(TMP.PERREDTRIBSAI,0) > 0 THEN
       UF_GET_INFORMACAO_ADICIONAL( TMP.IDEMPRESA , 'R', TMP.NCM, TMP.CODCEST)
        WHEN TMP.IDSITTRIBSAI = 41 THEN
                UF_GET_INFORMACAO_ADICIONAL( TMP.IDEMPRESA , 'N', TMP.NCM, TMP.CODCEST)
        WHEN TMP.IDSITTRIBSAI = 61 THEN
                 (
                        SELECT
                                          CIAA.CODIGOINFOADICIONAL
                        FROM
                                          DBA.CONFIG_INFORMACAO_ADICIONAL_AUTOMATICA AS CIAA
                        LEFT JOIN
                                          DBA.CONFIG_INFORMACAO_ADICIONAL_AUTOMATICA_NCM AS CIAAN
                                                                 ON ( CIAA.IDINFADIAUTO = CIAAN.IDINFADIAUTO )
                        WHERE
                                          CIAA.IDEMPRESA = TMP.IDEMPRESA AND
                                          CIAA.IDCFOP = 5656 AND
                                          CIAA.TIPOSITTRIB = 'M' AND
                                          CIAA.FLAGENVIACODIGOBENEFICIO = 'T' AND
                                          ( CIAAN.NCM = TMP.NCM OR TRIM( CIAAN.NCM ) = TMP.NCM OR CIAAN.NCM IS NULL )
                        FETCH FIRST 1 ROWS ONLY

    END) AS INFORMACAO_ADICIONAL,
    TMP.FLAGCONFERENCIAFRCX,
    DBA.UF_UTF8_TO_WIN1252( LEFT(TMP.DESCRICAOCOMPLETA,150)) AS DESCRICAOCOMPLETA,
    TMP.PERDESCONTOMAXIMO,
    TMP.FLAGCOMBUSTIVEL,
    TMP.GRAMATURA,
    TMP.IDCODIGOANP,
    DBA.UF_UTF8_TO_WIN1252( LEFT(TMP.DESCRICAOANP, 200)) AS DESCRICAOANP,
    TMP.PERGASNATURALGLP,
    TMP.PERGASNATURALGLPIMP,
    TMP.QTDMAXVENDAMES,
        (SELECT P.PATHTHUMBNAIL FROM PRODUTO_GRADE_IMAGENS P
          WHERE P.IDPRODUTO = TMP.IDPRODUTO AND P.IDSUBPRODUTO = TMP.IDSUBPRODUTO AND
           COALESCE(P.PATHTHUMBNAIL, '') <> ''
              ORDER BY P.FLAGIMAGEMPADRAO DESC, P.IDCODIMAGEM FETCH FIRST 1 ROWS ONLY) AS PATHTHUMBNAIL,
    TMP.REFERENCIA,
    TMP.PERIPISAIDA,
    TMP.FLAGBLOQUEIASEMESCANEAR,
    TMP.CLASSFISCAL,
    TMP.VALPRECOATACADO,
    TMP.GRAMATURAVENDA,
    TMP.QTDTOTALEMBALAGEM,
    TMP.PESOLIQUIDO,
    TMP.FLAGPERMITEDESCONTO,
    TMP.TIPOITEM,
    TMP.IDMARCAFABRICANTE,
    TMP.MODELO,
    TMP.FLAGRETENCAOPIS,
    TMP.FLAGRETENCAOCOFINS,
    TMP.FLAGRETENCAOIR,
    TMP.FLAGRETENCAOCONTRSOCIAL,
    TMP.IDMSG
FROM
    (
    SELECT
        PRODUTO_TRIBUTACAO_VIEW.IDPRODUTO,
        PRODUTO_TRIBUTACAO_VIEW.IDSUBPRODUTO,
        PRODUTO_TRIBUTACAO_VIEW.PERICMSAI,
        PRODUTO_TRIBUTACAO_VIEW.TIPOSITTRIBSAI,
        PRODUTO_TRIBUTACAO_VIEW.IDSITTRIBSAI,
        PRODUTO_TRIBUTACAO_VIEW.PERREDTRIBSAI,
        PRODUTO_TRIBUTACAO_VIEW.PERREDSUBTRIB,
        PRODUTO_TRIBUTACAO_VIEW.PERICMSUBST,
        PRODUTO_TRIBUTACAO_VIEW.PERICMECF,
        PRODUTO_TRIBUTACAO_VIEW.FLAGISS,
        PRODUTO_TRIBUTACAO_VIEW.UFORIGEM,
        COALESCE(POLITICA_PRECO_PRODUTO.VALPRECOVAREJO,0) AS VALPRECOPRODUTO,
        CAST( CAST(COALESCE(POLITICA_PRECO_PRODUTO.DTINIPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 00:00:00' AS TIMESTAMP) AS DTINIPROMOCAO,
        CAST( CAST(COALESCE(POLITICA_PRECO_PRODUTO.DTFIMPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 23:59:59' AS TIMESTAMP) AS DTFIMPROMOCAO,
        POLITICA_PRECO_PRODUTO.VALPROMVAREJO AS VALPRECOPROMOCAO,
        POLITICA_PRECO_PRODUTO.IDEMPRESA,
        POLITICA_PRECO_PRODUTO.QTDMINPROMVAREJO,
        'F' AS FLAGPROMOCAO,
        PRODUTO_GRADE.IDCODBARPROD,
        PRODUTO_GRADE.DESCRRESPRODUTO,
        COALESCE(PRODUTO_GRADE.VALMULTIVENDAS, 1) AS VALMULTIVENDAS,
        PRODUTO_GRADE.IDINFORMACAONUTRICIONAL,
        PRODUTO_GRADE.FLAGCESTABASICA,
        PRODUTO_GRADE.FLAGPRODUTOACABADO,
        PRODUTO_GRADE.NCM,
        PRODUTO_GRADE.IDSUBCATIPIRECEITA,
        PRODUTO_GRADE.IDLISTASERVICORECEITA,
        PRODUTO_GRADE.FLAGCOMPONENTE,
        PRODUTO_GRADE.IDPRODUTOVASILHAME,
        PRODUTO_GRADE.IDSUBPRODUTOVASILHAME,
        PRODUTO_GRADE.FLAGEXPBALANCAGRADE AS FLAGEXPBALANCA,
        PRODUTO.TIPOBAIXAMESTRE,
        PRODUTO_GRADE.FLAGINATIVO,
        PRODUTO_GRADE.FLAGVASILHAME,
        PRODUTO.EMBALAGEMSAIDA,
        PRODUTO.VALGRAMASAIDA,
        PRODUTO.IDSECAO,
        PRODUTO.IDGRUPO,
        PRODUTO.IDSUBGRUPO,
        PRODUTO.IDDIVISAO,
        CASE WHEN PRODUTO_GRADE.DIASVALIDADE > 0 THEN
            PRODUTO_GRADE.DIASVALIDADE
        ELSE
            PRODUTO.DIASVALIDADE
        END AS DIASVALIDADE,
        PRODUTO_GRADE.FLAGBLOQUEIAVENDA,
        CASE WHEN PRODUTO_TRIBUTACAO_VIEW.FLAGEVIDENCIA = 'T' OR POLITICA_PRECO_PRODUTO.FLAGEVIDENCIA = 'T' THEN
            'T'
        ELSE
            'F'
        END AS FLAGEVIDENCIA,
        POLITICA_PRECO_PRODUTO.PERCARGATRIBUTARIA,
        COALESCE(CAST(POLITICA_PRECO_PRODUTO.PERALIQNACFEDERAL  AS NUMERIC(5,2)), 0) AS PERALIQFEDERAL,
        COALESCE(CAST(POLITICA_PRECO_PRODUTO.PERALIQESTADUAL  AS NUMERIC(5,2)), 0) AS PERALIQESTADUAL,
        COALESCE(CAST(POLITICA_PRECO_PRODUTO.PERALIQMUNICIPAL AS NUMERIC(5,2)), 0) AS PERALIQMUNICIPAL,
        COALESCE(CAST(UF_GETCSTPISCOFINS( POLITICA_PRECO_PRODUTO.IDEMPRESA , 1300, PRODUTO_GRADE.IDSUBPRODUTO,'S', NULL) AS INTEGER), 1) AS IDCSTPISCOFINS,
        DBA.MAX_COLUMN_TIMESTAMP(
            DBA.MAX_COLUMN_TIMESTAMP(
                DBA.MAX_COLUMN_TIMESTAMP(PRODDTALTERACAO, PRODGRADEDTALTERACAO),
                DBA.MAX_COLUMN_TIMESTAMP(GRUPODTALTERACAO, POLITICA_PRECO_PRODUTO.DTALTERACAO)
            ),
            DBA.MAX_COLUMN_TIMESTAMP(
                DBA.MAX_COLUMN_TIMESTAMP(PRODTRIBESTDTALTERACAO, GRUPOTRIBESTDTALTERACAO),
                DBA.MAX_COLUMN_TIMESTAMP(PRODTRIBDTALTERACAO, GRUPOTRIBDTALTERACAO)
            )
        ) AS DTALTERACAO,
        PRODUTO_GRADE.IDSUBPRODUTOEXTERNO,
        CAST(NULL AS INTEGER) AS IDCENARIOFISCAL,
        PRODUTO_GRADE.CODCEST,
        PRODUTO.FLAGSOLICITAPESAGEM,
        COALESCE(PRODUTO_TRIBUTACAO_VIEW.PERFCEP, 0) AS PERFCEP,
        PRODUTO_GRADE.PESOBRUTO,
        PRODUTO_GRADE.FLAGPRODUZIDOESCALANAORELEVANTE,
        PRODUTO.IDMARCAFABRICANTE,
        PRODUTO_GRADE.IDCADEIAPRECO,
        PRODUTO_GRADE.IDCODBARPRODTRIB,
        PRODUTO_GRADE.FLAGLOTE,
        PRODUTO_GRADE.FLAGTINTA,
        PRODUTO_GRADE.FLAGCONFERENCIAFRCX,
        (PRODUTO.DESCRCOMPRODUTO || ' ' || COALESCE(PRODUTO_GRADE.SUBDESCRICAO,'')) AS DESCRICAOCOMPLETA,
                DESCONTO_MAXIMO_PRODUTO_EMP.PERDESCONTOMAXIMO,
                  COALESCE( PRODUTO_GRADE.FLAGCOMBUSTIVEL, 'F') AS FLAGCOMBUSTIVEL,
        COALESCE((SELECT GRAMATURA FROM PRODUTO_GRADE_UNIDADE_TRIBUTARIA PGUNT WHERE PGUNT.IDPRODUTO = PRODUTO_GRADE.IDPRODUTO AND PGUNT.IDSUBPRODUTO = PRODUTO_GRADE.IDSUBPRODUTO FETCH FIRST 1 ROWS ONLY), 0) AS GRAMATURA,
        COALESCE(IDCODIGOANP, '') AS IDCODIGOANP,
        CASE WHEN COALESCE(IDCODIGOANP, '') <> '' THEN (SELECT COALESCE(DESCRICAO, 'CODANP SEM DESCRICAO') FROM CODIGO_PRODUTO_ANP WHERE IDCODIGOANP = PRODUTO_GRADE.IDCODIGOANP FETCH FIRST 1 ROWS ONLY)  ELSE  '' END AS DESCRICAOANP,
        COALESCE(PERGASNATURALGLP, 0) as PERGASNATURALGLP,
        COALESCE(PERGASNATURALGLPIMP, 0) as PERGASNATURALGLPIMP,
        PRODUTO_GRADE.QTDMAXVENDAMES,
        COALESCE(PRODUTO_GRADE.REFERENCIA, '') AS REFERENCIA,
        PRODUTO_GRADE.PERIPIEQINDUSTRIAL AS PERIPISAIDA,
        PRODUTO_GRADE.FLAGBLOQUEIASEMESCANEAR AS FLAGBLOQUEIASEMESCANEAR,
        PRODUTO.CLASSFISCAL AS CLASSFISCAL,
        COALESCE(POLITICA_PRECO_PRODUTO.VALPRECOATACADO,0) AS VALPRECOATACADO,
        PRODUTO_GRADE.GRAMATURAVENDA,
        PRODUTO_GRADE.QTDTOTALEMBALAGEM,
        PRODUTO_GRADE.PESOLIQUIDO,
        PRODUTO_GRADE.FLAGPERMITEDESCONTO ,
        PRODUTO_GRADE.TIPOITEM,
        PRODUTO.MODELO,
		CASE WHEN PRODUTO.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOPIS
		ELSE
			PRODUTO.FLAGRETENCAOPIS
		END AS FLAGRETENCAOPIS,
		CASE WHEN PRODUTO.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOCOFINS
		ELSE
			PRODUTO.FLAGRETENCAOCOFINS
		END AS FLAGRETENCAOCOFINS,
		CASE WHEN PRODUTO.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOIR
		ELSE
			PRODUTO.FLAGRETENCAOIR
		END AS FLAGRETENCAOIR,
		CASE WHEN PRODUTO.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOCONTRSOCIAL
		ELSE
			PRODUTO.FLAGRETENCAOCONTRSOCIAL
		END AS FLAGRETENCAOCONTRSOCIAL,
         PRODUTO_TRIBUTACAO_VIEW.IDMSG
    FROM
        PRODUTOS
        JOIN DBA.PRODUTO_TRIBUTACAO_VIEW PRODUTO_TRIBUTACAO_VIEW ON (
            PRODUTO_TRIBUTACAO_VIEW.IDPRODUTO = PRODUTOS.IDPRODUTO AND
            PRODUTO_TRIBUTACAO_VIEW.IDSUBPRODUTO = PRODUTOS.IDSUBPRODUTO AND
            PRODUTO_TRIBUTACAO_VIEW.IDEMPRESA IN ( 0, PRODUTOS.IDEMPRESA ) AND
            PRODUTO_TRIBUTACAO_VIEW.UFORIGEM IN ( PRODUTOS.UF , '')
        )
        JOIN DBA.POLITICA_PRECO_PRODUTO POLITICA_PRECO_PRODUTO ON (
            POLITICA_PRECO_PRODUTO.IDPRODUTO = PRODUTOS.IDPRODUTO AND
            POLITICA_PRECO_PRODUTO.IDSUBPRODUTO = PRODUTOS.IDSUBPRODUTO AND
            POLITICA_PRECO_PRODUTO.IDEMPRESA = PRODUTOS.IDEMPRESA
        )
        JOIN DBA.PRODUTO_GRADE PRODUTO_GRADE ON (
            PRODUTO_GRADE.IDPRODUTO = PRODUTOS.IDPRODUTO AND
            PRODUTO_GRADE.IDSUBPRODUTO = PRODUTOS.IDSUBPRODUTO AND
           ('F' = :RA_COMPLETA OR PRODUTO_GRADE.FLAGINATIVO <>  :RA_COMPLETA OR (SELECT COALESCE(V_VARCHAR,'F') FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTOVALOR = (
                SELECT MAX(IDATRIBUTOVALOR) FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTO = 273 AND IDEMPRESA = POLITICA_PRECO_PRODUTO.IDEMPRESA)) = 'T')
        )
        JOIN DBA.PRODUTO PRODUTO ON (
            PRODUTO.IDPRODUTO = PRODUTOS.IDPRODUTO
        )
        JOIN EMPRESAS_BASE EMPRESA ON (
            EMPRESA.IDEMPRESA = POLITICA_PRECO_PRODUTO.IDEMPRESA
        )
        JOIN DBA.GRUPO GRUPO ON (
            GRUPO.IDGRUPO = PRODUTO.IDGRUPO
        )
      LEFT OUTER JOIN DBA.DESCONTO_MAXIMO_PRODUTO_EMP DESCONTO_MAXIMO_PRODUTO_EMP ON (
                    DESCONTO_MAXIMO_PRODUTO_EMP.IDEMPRESA = POLITICA_PRECO_PRODUTO.IDEMPRESA AND
                    DESCONTO_MAXIMO_PRODUTO_EMP.IDPRODUTO = PRODUTOS.IDPRODUTO AND
                    DESCONTO_MAXIMO_PRODUTO_EMP.IDSUBPRODUTO = PRODUTOS.IDSUBPRODUTO
      )

      LEFT JOIN LATERAL (
          SELECT DISTINCT
              'T' AS FLAGMIX,
              IDPRODUTO,
              IDSUBPRODUTO
          FROM
              DBA.PRODUTO_EMPRESA AS PRODUTO_EMPRESA
          WHERE
              PRODUTO_EMPRESA.IDEMPRESA = POLITICA_PRECO_PRODUTO.IDEMPRESA AND
              PRODUTO_GRADE.IDPRODUTO = PRODUTO_EMPRESA.IDPRODUTO AND
              PRODUTO_GRADE.IDSUBPRODUTO = PRODUTO_EMPRESA.IDSUBPRODUTO
      ) AS MIX ON (
          MIX.IDPRODUTO = PRODUTO_GRADE.IDPRODUTO AND
          MIX.IDSUBPRODUTO = PRODUTO_GRADE.IDSUBPRODUTO
      )
      LEFT JOIN DBA.PRODUTO_EMPRESA PE ON (
          PE.IDEMPRESA = POLITICA_PRECO_PRODUTO.IDEMPRESA       AND
          PE.IDPRODUTO    = POLITICA_PRECO_PRODUTO.IDPRODUTO    AND
          PE.IDSUBPRODUTO = POLITICA_PRECO_PRODUTO.IDSUBPRODUTO
      )

      LEFT JOIN
      ( SELECT IDEMPRESA, COALESCE(V_VARCHAR, 'F') AS FLAGPRODUTOEMPRESA
         FROM CONFIG_ATRIBUTO_VALORES
         WHERE IDATRIBUTO = 159
       ) AS CAV ON (CAV.IDEMPRESA = EMPRESA.IDEMPRESA)

    WHERE
       ( (CAV.FLAGPRODUTOEMPRESA = 'F' OR CAV.FLAGPRODUTOEMPRESA IS NULL) OR
         (CAV.FLAGPRODUTOEMPRESA = 'T' AND (MIX.FLAGMIX IS NULL OR PE.FLAGPERMITEVENDA = 'T' OR PE.FLAGPERMITEVENDA IS NULL))) AND
        PRODUTO.TIPOBAIXAMESTRE <> 'C' AND
        PRODUTO_GRADE.TIPOITEM NOT IN('07', '08')

    UNION ALL

    SELECT
        PG.IDPRODUTO,
        PG.IDSUBPRODUTO,
        CAST ('0' AS DECIMAL(16,2)) AS PERICMSAI,
        CAST ('I' AS CHAR) AS TIPOSITTRIBSAI,
        CAST ('40' AS INTEGER) AS IDSITTRIBSAI,
        CAST ('0' AS DECIMAL(16,2)) AS PERREDTRIBSAI,
        CAST ('0' AS DECIMAL(5,2)) AS PERREDSUBTRIB,
        CAST ('0' AS DECIMAL(5,2)) AS PERICMSUBST,
        CAST ('0' AS DECIMAL(16,2)) AS PERICMECF,
        CAST ('F' AS CHAR) AS FLAGISS,
        CAST (EMPRESA.UF AS CHAR(2)) AS UFORIGEM,
        COALESCE(PPP.VALPRECOVAREJO,0) AS VALPRECOPRODUTO,
        CAST( CAST(COALESCE(PPP.DTINIPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 00:00:00' AS TIMESTAMP) AS DTINIPROMOCAO,
        CAST( CAST(COALESCE(PPP.DTFIMPROMOCAOVAR,'1900-01-01') AS CHAR(10)) || ' 23:59:59' AS TIMESTAMP) AS DTFIMPROMOCAO,
        PPP.VALPROMVAREJO AS VALPRECOPROMOCAO,
        PPP.IDEMPRESA,
        PPP.QTDMINPROMVAREJO,
        'F' AS FLAGPROMOCAO,
        PG.IDCODBARPROD,
        PG.DESCRRESPRODUTO,
        COALESCE(PG.VALMULTIVENDAS, 1) AS VALMULTIVENDAS,
        PG.IDINFORMACAONUTRICIONAL,
        PG.FLAGCESTABASICA,
        PG.FLAGPRODUTOACABADO,
        PG.NCM,
        PG.IDSUBCATIPIRECEITA,
        PG.IDLISTASERVICORECEITA,
        PG.FLAGCOMPONENTE,
        PG.IDPRODUTOVASILHAME,
        PG.IDSUBPRODUTOVASILHAME,
        PG.FLAGEXPBALANCAGRADE AS FLAGEXPBALANCA,
        P.TIPOBAIXAMESTRE,
        COALESCE(PG.FLAGINATIVO, 'F') AS FLAGINATIVO,
        COALESCE(PG.FLAGVASILHAME, 'F') AS FLAGVASILHAME,
        P.EMBALAGEMSAIDA,
        P.VALGRAMASAIDA,
        P.IDSECAO,
        P.IDGRUPO,
        P.IDSUBGRUPO,
        P.IDDIVISAO,
        CASE WHEN PG.DIASVALIDADE > 0 THEN
            PG.DIASVALIDADE
        ELSE
            P.DIASVALIDADE
        END AS DIASVALIDADE,
        COALESCE(PG.FLAGBLOQUEIAVENDA, 'F') AS FLAGBLOQUEIAVENDA,
        CASE WHEN P.FLAGEVIDENCIA = 'T' OR PG.FLAGEVIDENCIA= 'T' OR PPP.FLAGEVIDENCIA = 'T' THEN
            'T'
        ELSE
            'F'
        END AS FLAGEVIDENCIA,
        PPP.PERCARGATRIBUTARIA,
        PPP.PERALIQFEDERAL AS PERALIQFEDERAL,
        PPP.PERALIQESTADUAL AS PERALIQESTADUAL,
        PPP.PERALIQMUNICIPAL AS PERALIQMUNICIPAL,
        COALESCE(CAST(UF_GETCSTPISCOFINS( PPP.IDEMPRESA , 1300, PG.IDSUBPRODUTO,'S', NULL) AS INTEGER), 1) AS IDCSTPISCOFINS,
        CASE WHEN P.DTALTERACAO > PG.DTALTERACAO AND P.DTALTERACAO > PPP.DTALTERACAO THEN
            P.DTALTERACAO
        WHEN PG.DTALTERACAO > PPP.DTALTERACAO THEN
            PG.DTALTERACAO
        ELSE
            PPP.DTALTERACAO
        END AS DTALTERACAO,
        PG.IDSUBPRODUTOEXTERNO,
        CAST ('0' AS INTEGER) AS IDCENARIOFISCAL,
        PG.CODCEST,
        P.FLAGSOLICITAPESAGEM,
        CAST ('0' AS DECIMAL(16,2)) AS PERFCEP,
        PG.PESOBRUTO,
        PG.FLAGPRODUZIDOESCALANAORELEVANTE,
        P.IDMARCAFABRICANTE,
        PG.IDCADEIAPRECO,
        PG.IDCODBARPRODTRIB,
        PG.FLAGLOTE,
        PG.FLAGTINTA,
        PG.FLAGCONFERENCIAFRCX,
        (P.DESCRCOMPRODUTO || ' ' || COALESCE(PG.SUBDESCRICAO,'')) AS DESCRICAOCOMPLETA,
                  DESCONTO_MAXIMO_PRODUTO_EMP.PERDESCONTOMAXIMO,
                  COALESCE( PG.FLAGCOMBUSTIVEL, 'F') AS FLAGCOMBUSTIVEL,
        COALESCE((SELECT GRAMATURA FROM PRODUTO_GRADE_UNIDADE_TRIBUTARIA PGUNT WHERE PGUNT.IDPRODUTO = PG.IDPRODUTO AND PGUNT.IDSUBPRODUTO = PG.IDSUBPRODUTO FETCH FIRST 1 ROWS ONLY), 0) AS GRAMATURA,
        COALESCE(IDCODIGOANP, '') AS IDCODIGOANP,
        CASE WHEN COALESCE(IDCODIGOANP, '') <> '' THEN (SELECT COALESCE(DESCRICAO, 'CODANP SEM DESCRICAO') FROM CODIGO_PRODUTO_ANP WHERE IDCODIGOANP = PG.IDCODIGOANP FETCH FIRST 1 ROWS ONLY)  ELSE  '' END AS DESCRICAOANP,
        COALESCE(PERGASNATURALGLP, 0) as PERGASNATURALGLP,
        COALESCE(PERGASNATURALGLPIMP, 0) as PERGASNATURALGLPIMP,
        PG.QTDMAXVENDAMES,
        COALESCE(PG.REFERENCIA, '') AS REFERENCIA,
        PG.PERIPIEQINDUSTRIAL AS PERIPISAIDA,
        PG.FLAGBLOQUEIASEMESCANEAR AS FLAGBLOQUEIASEMESCANEAR,
        P.CLASSFISCAL AS CLASSFISCAL,
        COALESCE(PPP.VALPRECOATACADO,0) AS VALPRECOATACADO,
        PG.GRAMATURAVENDA,
        PG.QTDTOTALEMBALAGEM,
        PG.PESOLIQUIDO,
        PG.FLAGPERMITEDESCONTO ,
        PG.TIPOITEM,
        P.MODELO,
		CASE WHEN P.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOPIS
		ELSE
			P.FLAGRETENCAOPIS
		END AS FLAGRETENCAOPIS,
		CASE WHEN P.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOCOFINS
		ELSE
			P.FLAGRETENCAOCOFINS
		END AS FLAGRETENCAOCOFINS,
		CASE WHEN P.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOIR
		ELSE
			P.FLAGRETENCAOIR
		END AS FLAGRETENCAOIR,
		CASE WHEN P.FLAGTRIBUTACAOGRUPO = 'T' THEN
			GRUPO.FLAGRETENCAOCONTRSOCIAL
		ELSE
			P.FLAGRETENCAOCONTRSOCIAL
		END AS FLAGRETENCAOCONTRSOCIAL,
        CAST(NULL AS INTEGER) AS IDMSG
    FROM
        DBA.POLITICA_PRECO_PRODUTO PPP
        JOIN DBA.PRODUTO_GRADE PG ON (
            PG.IDPRODUTO = PPP.IDPRODUTO AND
            PG.IDSUBPRODUTO = PPP.IDSUBPRODUTO
        )
        JOIN DBA.PRODUTO P ON (
            P.IDPRODUTO = PG.IDPRODUTO
        )
        JOIN DBA.GRUPO GRUPO ON (
            GRUPO.IDGRUPO = P.IDGRUPO
        )
        JOIN EMPRESAS_BASE EMPRESA ON (
            EMPRESA.IDEMPRESA = PPP.IDEMPRESA
        )
        LEFT OUTER JOIN DBA.DESCONTO_MAXIMO_PRODUTO_EMP DESCONTO_MAXIMO_PRODUTO_EMP ON (
                                          DESCONTO_MAXIMO_PRODUTO_EMP.IDEMPRESA = PPP.IDEMPRESA AND
                                          DESCONTO_MAXIMO_PRODUTO_EMP.IDPRODUTO = PPP.IDSUBPRODUTO AND
                                          DESCONTO_MAXIMO_PRODUTO_EMP.IDSUBPRODUTO = PPP.IDSUBPRODUTO
        )
        LEFT JOIN LATERAL (
            SELECT DISTINCT
                'T' AS FLAGMIX,
                IDPRODUTO,
                IDSUBPRODUTO
            FROM
                DBA.PRODUTO_EMPRESA AS PRODUTO_EMPRESA
            WHERE
                PRODUTO_EMPRESA.IDEMPRESA = PPP.IDEMPRESA AND
                PG.IDPRODUTO = PRODUTO_EMPRESA.IDPRODUTO AND
                PG.IDSUBPRODUTO = PRODUTO_EMPRESA.IDSUBPRODUTO
        ) AS MIX ON (
            MIX.IDPRODUTO = PG.IDPRODUTO AND
            MIX.IDSUBPRODUTO = PG.IDSUBPRODUTO
        )
        LEFT JOIN DBA.PRODUTO_EMPRESA PE ON (
            PE.IDEMPRESA = PPP.IDEMPRESA       AND
            PE.IDPRODUTO    = PPP.IDPRODUTO    AND
            PE.IDSUBPRODUTO = PPP.IDSUBPRODUTO
        )
        LEFT JOIN
        ( SELECT IDEMPRESA, COALESCE(V_VARCHAR, 'F') AS FLAGPRODUTOEMPRESA
          FROM CONFIG_ATRIBUTO_VALORES
          WHERE IDATRIBUTO = 159
        ) AS CAV ON (CAV.IDEMPRESA = EMPRESA.IDEMPRESA)

    WHERE
        P.TIPOBAIXAMESTRE = 'C' AND
        ('F' = :RA_COMPLETA OR PG.FLAGINATIVO <> :RA_COMPLETA OR
                        (SELECT COALESCE(V_VARCHAR, 'F') FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTOVALOR = (
                SELECT MAX(IDATRIBUTOVALOR) FROM CONFIG_ATRIBUTO_VALORES WHERE IDATRIBUTO = 273 AND IDEMPRESA = PPP.IDEMPRESA)) = 'T') AND
        (
            COALESCE(PG.DTALTERACAO, CAST('1900-01-01 00:00:00' AS TIMESTAMP)) >= :RA_DTMONITOR OR
            COALESCE(P.DTALTERACAO, CAST('1900-01-01 00:00:00' AS TIMESTAMP)) >= :RA_DTMONITOR OR
            COALESCE(PPP.DTALTERACAO, CAST('1900-01-01 00:00:00' AS TIMESTAMP)) >= :RA_DTMONITOR
        ) AND
        ( (CAV.FLAGPRODUTOEMPRESA = 'F' OR CAV.FLAGPRODUTOEMPRESA IS NULL) OR
        (CAV.FLAGPRODUTOEMPRESA = 'T' AND (MIX.FLAGMIX IS NULL OR PE.FLAGPERMITEVENDA = 'T' OR PE.FLAGPERMITEVENDA IS NULL))) AND
        PG.TIPOITEM NOT IN('07', '08')

    ) AS TMP
LEFT OUTER JOIN DBA.MARCA MARCA ON
(
        MARCA.IDMARCAFABRICANTE = TMP.IDMARCAFABRICANTE
    )
